//Criação da API  INDEED
//A API é apenas o exemplo de como funcionario

fun main(args: Array<String>) {

    //Candidato busca a vaga por essas palavras chaves:
    //Criação de classes,
    //titulo - representa o nome do trabalho
    //empresa - nome da empresa
    //local - localização do trabalho
    //url - o link da vaga no indeed

    data class emprego(
        val titulo: String,
        val empresa: String,
        val local: String,
        val url: String
    )

    //A API retorna o conteudo de acordo com a palavra chave
    fun fetchJobsFromIndeedAPI(): List<emprego> {
        return listOf(
            emprego(
                titulo = "Desenvolvedor Android",
                empresa = "Empresa A",
                local = "São Paulo, SP",
                url = "https://www.indeed.com/job/12345"
            ),
            emprego(
                titulo = "Analista de Dados",
                empresa = "Empresa B",
                local = "Remoto",
                url = "https://www.indeed.com/job/67890"
            ),
            emprego(
                titulo = "Designer UX/UI",
                empresa = "Empresa C",
                local = "Rio de Janeiro, RJ",
                url = "https://www.indeed.com/job/54321"
            )
        )
    }

    // A função simula a API e retonar os dados no console
    fun main() {
        val emprego = fetchJobsFromIndeedAPI()
        println("Vagas de Emprego:")
        emprego.forEach { emprego ->
            println("Título: ${emprego.titulo}")
            println("Empresa: ${emprego.empresa}")
            println("Localização: ${emprego.local}")
            println("URL: ${emprego.url}")
            println("-----------------------------")
        }
    }

}
